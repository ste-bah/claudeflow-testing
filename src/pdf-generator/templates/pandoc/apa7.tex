%% Page style for running head
\pagestyle{fancy}
\fancyhf{}

%% Running head: ALL CAPS flush left, page number flush right
%% Per APA 7th Edition Section 2.18
$if(running-head)$
\fancyhead[L]{\footnotesize\MakeUppercase{$running-head$}}
$else$
\fancyhead[L]{\footnotesize\MakeUppercase{$title$}}
$endif$
\fancyhead[R]{\thepage}
\renewcommand{\headrulewidth}{0pt}

%% Plain style for first page of sections (still has running head in APA 7)
\fancypagestyle{plain}{
  \fancyhf{}
  $if(running-head)$
  \fancyhead[L]{\footnotesize\MakeUppercase{$running-head$}}
  $else$
  \fancyhead[L]{\footnotesize\MakeUppercase{$title$}}
  $endif$
  \fancyhead[R]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

%% =============================================================================
%% HEADING STYLES - APA 7th Edition Section 2.27
%% =============================================================================

%% Level 1: Centered, Bold, Title Case
%% Spacing: 1 blank double-spaced line above (12pt), 0pt below
\titleformat{\section}
  {\normalfont\normalsize\bfseries\centering}
  {}
  {0pt}
  {}
\titlespacing*{\section}{0pt}{\baselineskip}{0pt}

%% Level 2: Flush Left, Bold, Title Case
\titleformat{\subsection}
  {\normalfont\normalsize\bfseries\raggedright}
  {}
  {0pt}
  {}
\titlespacing*{\subsection}{0pt}{\baselineskip}{0pt}

%% Level 3: Flush Left, Bold Italic, Title Case
\titleformat{\subsubsection}
  {\normalfont\normalsize\bfseries\itshape\raggedright}
  {}
  {0pt}
  {}
\titlespacing*{\subsubsection}{0pt}{\baselineskip}{0pt}

%% Level 4: Indented, Bold, Title Case, Ending with Period.
\titleformat{\paragraph}[runin]
  {\normalfont\normalsize\bfseries}
  {}
  {0pt}
  {\hspace{\parindent}}
  [.]
\titlespacing*{\paragraph}{0pt}{\baselineskip}{\wordsep}

%% Level 5: Indented, Bold Italic, Title Case, Ending with Period.
\titleformat{\subparagraph}[runin]
  {\normalfont\normalsize\bfseries\itshape}
  {}
  {0pt}
  {\hspace{\parindent}}
  [.]
\titlespacing*{\subparagraph}{0pt}{\baselineskip}{\wordsep}
